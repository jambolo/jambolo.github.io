252 void Bitmap<_Pixel>::Draw(Rect const & srcRect, Bitmap * pDst, int pDx, int pDy) const
253 {
254  Rect rect = srcRect;
255 
256  ClipForDrawing(&rect, &pDx, &pDy, pDst->GetWidth(), pDst->GetHeight());
257 
258  if (rect.m_Width <= 0 || rect.m_Height <= 0)
259  return;
260 
261  Pixel * dstp = pDst->GetMap(pDx, pDy);
262  Pixel const * srcp = GetMap(rect.m_X, rect.m_Y);
263  int const dpitch = pDst->GetPitch();
264 
265  for (int i = 0; i < rect.m_Height; i++)
266  {
267  memcpy(dstp, srcp, rect.m_Width * PIXEL_SIZE);
268 
269  dstp = reinterpret_cast<Pixel *>(reinterpret_cast<char *>(dstp) + dpitch);
270  srcp = reinterpret_cast<Pixel const *>(reinterpret_cast<char const *>(srcp) + m_Pitch);
271  }
272 }

274 template <class _Pixel>
275 void Bitmap<_Pixel>::ClipForDrawing(Rect * srcRect, int * pDx, int * pDy, int dw, int dh) const
276 {
277  // Clip to source bitmap
278 
279  if (srcRect->m_X < 0)
280  {
281  srcRect->m_Width += srcRect->m_X;
282  pDx -= srcRect->m_X;
283  srcRect->m_X = 0;
284  }
285 
286  if (srcRect->m_Y < 0)
287  {
288  srcRect->m_Height += srcRect->m_Y;
289  pDy -= srcRect->m_Y;
290  srcRect->m_Y = 0;
291  }
292 
293  if (srcRect->m_Width > m_Width - srcRect->m_X)
294  {
295  srcRect->m_Width = m_Width - srcRect->m_X;
296  }
297 
298  if (srcRect->m_Height > m_Height - srcRect->m_Y)
299  {
300  srcRect->m_Height = m_Height - srcRect->m_Y;
301  }
302 
303  // Clip to dest bitmap
304 
305  int ldx = *pDx;
306  int ldy = *pDy;
307 
308  if (ldx < 0)
309  {
310  srcRect->m_Width += ldx;
311  srcRect->m_X -= ldx;
312  ldx = 0;
313  }
314 
315  if (ldy < 0)
316  {
317  srcRect->m_Height += ldy;
318  srcRect->m_X -= ldy;
319  ldy = 0;
320  }
321 
322  *pDx = ldx;
323  *pDy = ldy;
324 
325  if (srcRect->m_Width > dw - ldx)
326  {
327  srcRect->m_Width = dw - ldx;
328  }
329 
330  if (srcRect->m_Height > dh - ldy)
331  {
332  srcRect->m_Height = dh - ldy;
333  }
334 }
